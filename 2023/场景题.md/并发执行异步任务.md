function multiRequest(reqArr, limitNum) {
  const len = reqArr.length; // 请求总数量
  // 根据请求数量创建一个数组来保存请求的结果
  const res = new Array(len).fill(false);
  let count = 0; // 请求计数器
  return new Promise((resolve, reject) => {
    while (count < limitNum) {
      next();
    }
    async function next() {
      let cur = count++;
      // 当队列中的所有请求都处理时将结果resolve出去
      if (cur >= len) {
        return !res.includes(false) && resolve(res);
      }
      const fn = reqArr[cur];
      const data = await fn;
      res[cur] = data;
      // 递归执行 从reqArr取一个去补上
      // 满足最大并发数limitNum
      next();
    }
  });
}
